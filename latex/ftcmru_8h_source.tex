\DoxyCodeLine{21   \textcolor{comment}{/* An MRU is a list that cannot hold more than a certain number of       */}}
\DoxyCodeLine{22   \textcolor{comment}{/* elements (`max\_elements').  All elements in the list are sorted in    */}}
\DoxyCodeLine{23   \textcolor{comment}{/* least-\/recently-\/used order, i.e., the `oldest' element is at the tail  */}}
\DoxyCodeLine{24   \textcolor{comment}{/* of the list.                                                          */}}
\DoxyCodeLine{25   \textcolor{comment}{/*                                                                       */}}
\DoxyCodeLine{26   \textcolor{comment}{/* When doing a lookup (either through `Lookup()' or `Lookup\_Node()'),   */}}
\DoxyCodeLine{27   \textcolor{comment}{/* the list is searched for an element with the corresponding key.  If   */}}
\DoxyCodeLine{28   \textcolor{comment}{/* it is found, the element is moved to the head of the list and is      */}}
\DoxyCodeLine{29   \textcolor{comment}{/* returned.                                                             */}}
\DoxyCodeLine{30   \textcolor{comment}{/*                                                                       */}}
\DoxyCodeLine{31   \textcolor{comment}{/* If no corresponding element is found, the lookup routine will try to  */}}
\DoxyCodeLine{32   \textcolor{comment}{/* obtain a new element with the relevant key.  If the list is already   */}}
\DoxyCodeLine{33   \textcolor{comment}{/* full, the oldest element from the list is discarded and replaced by a */}}
\DoxyCodeLine{34   \textcolor{comment}{/* new one; a new element is added to the list otherwise.                */}}
\DoxyCodeLine{35   \textcolor{comment}{/*                                                                       */}}
\DoxyCodeLine{36   \textcolor{comment}{/* Note that it is possible to pre-\/allocate the element list nodes.      */}}
\DoxyCodeLine{37   \textcolor{comment}{/* This is handy if `max\_elements' is sufficiently small, as it saves    */}}
\DoxyCodeLine{38   \textcolor{comment}{/* allocations/releases during the lookup process.                       */}}
\DoxyCodeLine{39   \textcolor{comment}{/*                                                                       */}}
